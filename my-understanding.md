# 1. สร้างโปรเจกต์ (เลือก React และ JavaScript)
npm create vite@latest react-assignment

# 2. เข้าไปในโฟลเดอร์
cd my-react-assignment

# 3. ติดตั้ง Library พื้นฐาน
npm install

# 4. ติดตั้ง Library เสริมสำหรับใช้ใน project (Router และ Axios)
npm install react-router-dom axios

react คือ Library ที่ทำหน้าที่จัดการ "เส้นทาง" (Path) ของเว็บไซต์
Axios จะใช้สำหรับแปลงข้อมูลที่ได้จาก API ให้เป็น JSON

# 5. ติดตั้ง Package tailwind 
โดยทำตามขั้นตอนใน https://tailwindcss.com/docs/installation/using-vite
